6   package com.zirinna.himmunhedelmapeli.userinterface;
7   
8   import com.zirinna.himmunhedelmapeli.gamelogic.GameLogic;
9   import javafx.scene.canvas.GraphicsContext;
10  import javafx.scene.paint.Color;
11  import javafx.scene.text.Font;
12  
13  /**
14   * Handles UI stuff.
15   * @author zirinna
16   */
17  public abstract class UserInterface {
18   
19      /**
20       * Draws the UI.
21       * @param gc graphicscontext to draw the UI to.
22       * @param gl gamelogic
23       */
24      public static void drawUI(GraphicsContext gc, GameLogic gl) {
25          gc.save();
26          double screenWidth = gc.getCanvas().getWidth();
27          double screenHeight = gc.getCanvas().getHeight();
28          
29          gc.setFill(Color.DARKRED);
30          gc.setFont(Font.font("Comic Sans", 24));
31          gc.fillText("Pisteet: " + gl.getScore(), screenWidth - 150, 20);
32          gc.fillText("Siirtoja: " + gl.getMoves(), screenWidth - 150, 50);
33          
34          if (gl.getMoves() <= 0) {
35              drawGameOver(gc, gl);
36          }
37          
38          gc.restore();
39          
40      }
41      
42      /**
43       * Tells all the tiles on the board to draw themselves on the graphicscontext.
44       * @param gc graphicscontext to draw the tiles on.
45       * @param gamelogic the gamelogic to take the tiles from
46       */
47      public static void drawTiles(GraphicsContext gc, GameLogic gamelogic) {
48          for (int x = 0; x < gamelogic.getGameBoard().getBoardSize(); x++) {
49              for (int y = 0; y < gamelogic.getGameBoard().getBoardSize(); y++) {
50                  gamelogic.getGameBoard().getTile(x, y).drawTile(gc);
51              }
52          }
53      }
54      
55      /**
56       * Draws the game over -pop up.
57       * @param gc graphicscontext to draw on
58       * @param gl gamelogic
59       */
60      public static void drawGameOver(GraphicsContext gc, GameLogic gl) {
61          gc.setFill(Color.DARKCYAN);
62          gc.fillRect(50, 50, 200, 125);
63          gc.setFill(Color.WHITE);
64          gc.setFont(Font.font("Comic Sans", 24));
65          gc.fillText("Peli loppui, sait ", 75, 85);
66          gc.fillText("" + gl.getScore(), 115, 110);
67          gc.fillText("pistett√§.", 90, 135);
68      }
69  }
