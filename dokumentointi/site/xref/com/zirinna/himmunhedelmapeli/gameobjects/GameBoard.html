6   package com.zirinna.himmunhedelmapeli.gameobjects;
7   
8   /**
9    * This class is used to represent the gameboard, which holds the tiles.
10   * @author zirinna
11   */
12  public class GameBoard {
13      
14      private Tile[][] gameboard;
15      
16      public GameBoard(int boardSize) {
17          this.gameboard = new Tile[boardSize][boardSize];
18          generateBoard();
19      }
20      
21      private void generateBoard() {
22          for (int x = 0; x < this.gameboard.length; x++) {
23              for (int y = 0; y < this.gameboard.length; y++) {
24                  gameboard[x][y] = new Tile(x, y);
25              }
26          }
27      }
28      
29      public Tile getTile(int x, int y) {
30          return gameboard[x][y];
31      }
32      
33      public void setTile(int x, int y, Tile tile) {
34          this.gameboard[x][y] = tile;
35      }
36      
37      public int getBoardSize() {
38          return this.gameboard.length;
39      }
40  }
